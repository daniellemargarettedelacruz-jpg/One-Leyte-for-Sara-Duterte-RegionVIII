function verifyMember_(e) { const memberId = normalizeMemberId_(e.parameter.memberId || ""); const fullName = normalizeLookupText_(e.parameter.fullName || ""); if (!memberId || !fullName) { return { success: false, message: "MemberID and Full Name are required." }; } const sh = getMemberLookupSheet_(); const values = sh.getDataRange().getValues(); if (values.length < 2) { return { success: false, message: "No member records found." }; } const headers = values[0]; const rows = values.slice(1); const memberIdCol = findHeaderIndex_(headers, ["MemberID", "Member ID", "ID"]); const fullNameCol = findHeaderIndex_(headers, ["FullName", "Full Name", "Name"]); if (memberIdCol === -1) throw new Error("MemberID column not found."); if (fullNameCol === -1) throw new Error("FullName column not found."); for (let i = 0; i < rows.length; i++) { const rowMemberId = normalizeMemberId_(rows[i][memberIdCol] || ""); const rowFullName = normalizeLookupText_(rows[i][fullNameCol] || ""); if (rowMemberId === memberId && rowFullName === fullName) { return { success: true, memberId: rowMemberId, memberName: rows[i][fullNameCol], message: "Member verified successfully." }; } } return { success: false, message: "Member not found or Full Name does not match." }; }
